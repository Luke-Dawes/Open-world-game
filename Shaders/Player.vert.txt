#version 330 core

layout (location = 0) in vec3 aPos;   //position
layout (location = 1) in vec4 aColor; //color

out vec4 ourColor; // pass color to fragment shader


uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
uniform bool angry;

void main()
{
    gl_Position = projection * view * model * vec4(aPos, 1.0);
    if (angry) 
{
    ourColor = vec4(1.0, 0.0, 0.0, 1.0); //if angry set to red
}
    else 
{
    ourColor = aColor; //set to other colour
}
}
